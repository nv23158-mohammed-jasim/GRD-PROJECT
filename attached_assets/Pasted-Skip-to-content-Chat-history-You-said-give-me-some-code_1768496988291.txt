const form = document.getElementById("workout-form");
const tableBody = document.querySelector("#workout-table tbody");

// Load saved workouts
let workouts = JSON.parse(localStorage.getItem("workouts")) || [];
renderTable();

form.addEventListener("submit", (e) => {
  e.preventDefault();

  const exercise = document.getElementById("exercise").value;
  const duration = document.getElementById("duration").value;
  const calories = document.getElementById("calories").value;
  const date = new Date().toLocaleDateString();

  const workout = { exercise, duration, calories, date };

  workouts.push(workout);
  localStorage.setItem("workouts", JSON.stringify(workouts));

  renderTable();
  form.reset();
});

function renderTable() {
  tableBody.innerHTML = "";
  workouts.forEach((workout, index) => {
    const row = document.createElement("tr");
    row.innerHTML = `
      <td>${workout.exercise}</td>
      <td>${workout.duration} min</td>
      <td>${workout.calories} cal</td>
      <td>${workout.date}</td>
      <td><button class="delete-btn" onclick="deleteWorkout(${index})">Delete</button></td>
    `;
    tableBody.appendChild(row);
  });
}

function deleteWorkout(index) {
  workouts.splice(index, 1);
  localStorage.setItem("workouts", JSON.stringify(workouts));
  renderTable();
}

const video = document.getElementById("video");
const canvas = document.getElementById("canvas");
const ctx = canvas.getContext("2d");
const counterEl = document.getElementById("count");

let detector, poses;
let squatCount = 0;
let squatDown = false;

// Load camera
async function setupCamera() {
  video.width = 640;
  video.height = 480;

  const stream = await navigator.mediaDevices.getUserMedia({
    video: true
  });
  video.srcObject = stream;

  return new Promise((resolve) => {
    video.onloadedmetadata = () => {
      resolve(video);
    };
  });
}

// Load MoveNet model
async function loadModel() {
  detector = await poseDetection.createDetector(
    poseDetection.SupportedModels.MoveNet,
    { modelType: poseDetection.movenet.modelType.SINGLEPOSE_LIGHTNING }
  );
}

// Calculate angle between 3 points
function calculateAngle(a, b, c) {
  const ab = [a.x - b.x, a.y - b.y];
  const cb = [c.x - b.x, c.y - b.y];

  const dot = ab[0] * cb[0] + ab[1] * cb[1];
  const magAB = Math.sqrt(ab[0] ** 2 + ab[1] ** 2);
  const magCB = Math.sqrt(cb[0] ** 2 + cb[1] ** 2);

  const cosine = dot / (magAB * magCB);
  return Math.acos(cosine) * (180 / Math.PI);
}

// Track poses and count squats
async function detectPose() {
  poses = await detector.estimatePoses(video);

  ctx.clearRect(0, 0, canvas.width, canvas.height);
  ctx.drawImage(video, 0, 0, canvas.width, canvas.height);

  if (poses.length > 0) {
    const keypoints = poses[0].keypoints;

    // Get hip, knee, ankle
    const leftHip = keypoints.find(k => k.name === "left_hip");
    const leftKnee = keypoints.find(k => k.name === "left_knee");
    const leftAnkle = keypoints.find(k => k.name === "left_ankle");

    if (leftHip && leftKnee && leftAnkle) {
      const angle = calculateAngle(leftHip, leftKnee, leftAnkle);

      // Squat detection: going down if angle < 90
      if (angle < 90 && !squatDown) {
        squatDown = true;
      }
      // Standing back up (angle > 160) → count rep
      if (angle > 160 && squatDown) {
        squatCount++;
        squatDown = false;
        counterEl.textContent = squatCount;
      }
    }

    // Draw keypoints
    keypoints.forEach(point => {
      if (point.score > 0.4) {
        ctx.beginPath();
        ctx.arc(point.x, point.y, 5, 0, 2 * Math.PI);
        ctx.fillStyle = "red";
        ctx.fill();
      }
    });
  }

  requestAnimationFrame(detectPose);
}

async function init() {
  await setupCamera();
  await loadModel();
  detectPose();
}

// Calculate angle between 3 points
function calculateAngle(a, b, c) {
  const ab = [a.x - b.x, a.y - b.y];
  const cb = [c.x - b.x, c.y - b.y];

  const dot = ab[0] * cb[0] + ab[1] * cb[1];
  const magAB = Math.sqrt(ab[0] ** 2 + ab[1] ** 2);
  const magCB = Math.sqrt(cb[0] ** 2 + cb[1] ** 2);

  const cosine = dot / (magAB * magCB);
  return Math.acos(cosine) * (180 / Math.PI);
}

// Track poses and count squats
async function detectPose() {
  poses = await detector.estimatePoses(video);

  ctx.clearRect(0, 0, canvas.width, canvas.height);
  ctx.drawImage(video, 0, 0, canvas.width, canvas.height);

  if (poses.length > 0) {
    const keypoints = poses[0].keypoints;

    // Get hip, knee, ankle
    const leftHip = keypoints.find(k => k.name === "left_hip");
    const leftKnee = keypoints.find(k => k.name === "left_knee");
    const leftAnkle = keypoints.find(k => k.name === "left_ankle");

    if (leftHip && leftKnee && leftAnkle) {
      const angle = calculateAngle(leftHip, leftKnee, leftAnkle);

      // Squat detection: going down if angle < 90
      if (angle < 90 && !squatDown) {
        squatDown = true;
      }
      // Standing back up (angle > 160) → count rep
      if (angle > 160 && squatDown) {
        squatCount++;
        squatDown = false;
        counterEl.textContent = squatCount;
      }
    }

    // Draw keypoints
    keypoints.forEach(point => {
      if (point.score > 0.4) {
        ctx.beginPath();
        ctx.arc(point.x, point.y, 5, 0, 2 * Math.PI);
        ctx.fillStyle = "red";
        ctx.fill();
      }
    });

let detector = null;
let rafId = null;
let poses = [];

// rep/state variables
let repCount = 0;
let state = {}; // per-exercise state machine

// Make canvas match video size after metadata
function setCanvasSize() {
  canvas.width = video.videoWidth || 640;
  canvas.height = video.videoHeight || 480;
}

// --- Pose utilities ---
function calculateAngle(a, b, c) {
  // a,b,c are {x,y}
  const abx = a.x - b.x, aby = a.y - b.y;
  const cbx = c.x - b.x, cby = c.y - b.y;
  const dot = abx * cbx + aby * cby;
  const magAB = Math.hypot(abx, aby);
  const magCB = Math.hypot(cbx, cby);
  if (magAB === 0 || magCB === 0) return 0;
  const cos = Math.min(1, Math.max(-1, dot / (magAB * magCB)));
  return Math.acos(cos) * (180 / Math.PI);
}

// distance between two points
function dist(a, b) {
  return Math.hypot(a.x - b.x, a.y - b.y);
}

// Keypoints accessor with fallback by name or index
function kp(pose, name) {
  if (!pose) return null;
  // pose.keypoints may have either .name (MoveNet) or .part (PoseNet)
  return pose.keypoints.find(k => (k.name === name) || (k.part === name)) || null;
}

// draw improved skeleton & keypoints
function drawPose(pose) {
  const keypoints = pose.keypoints;
  // draw semi-transparent video was drawn already
  // joints pairs (simple skeleton)
  const pairs = [
    ["left_shoulder","right_shoulder"],
    ["left_shoulder","left_elbow"],
    ["left_elbow","left_wrist"],
    ["right_shoulder","right_elbow"],
    ["right_elbow","right_wrist"],
    ["left_shoulder","left_hip"],
    ["right_shoulder","right_hip"],
    ["left_hip","right_hip"],
    ["left_hip","left_knee"],
    ["left_knee","left_ankle"],
    ["right_hip","right_knee"],
    ["right_knee","right_ankle"]
  ];

  ctx.lineWidth = 3;
  ctx.strokeStyle = "rgba(179,0,0,0.9)";
  pairs.forEach(([a,b])=>{
    const A = keypoints.find(k=>k.name===a || k.part===a);
    const B = keypoints.find(k=>k.name===b || k.part===b);
    if (A && B && A.score>0.35 && B.score>0.35){
      ctx.beginPath();
      ctx.moveTo(A.x, A.y);
      ctx.lineTo(B.x, B.y);
      ctx.stroke();
    }
  });

  // draw cleaned keypoints (outer circle + inner small white)
  keypoints.forEach(p=>{
    if (!p || p.score < 0.35) return;
    ctx.beginPath();
    ctx.fillStyle = "rgba(179,0,0,0.85)";
    ctx.arc(p.x, p.y, 5, 0, Math.PI*2);
    ctx.fill();

    ctx.beginPath();
    ctx.fillStyle = "#fff";
    ctx.arc(p.x, p.y, 2.5, 0, Math.PI*2);
    ctx.fill();
  });
}

// --- Audio feedback (speech) ---
function speak(text) {
  if (!("speechSynthesis" in window)) return;
  const utter = new SpeechSynthesisUtterance(text);
  utter.lang = "en-US";
  utter.rate = 1;
  window.speechSynthesis.cancel(); // cancel previous for immediate feedback
  window.speechSynthesis.speak(utter);
}

// --- Exercise detection logic ---
function resetStateForExercise(ex) {
  state = {
    lastAngle: null,
    down: false,
    up: false,
    lastDistance: null,
    lastTimeCue: 0
  };
  repCount = 0;
  countEl.textContent = repCount;
  cueEl.textContent = "Ready";
}

function handleSquats(pose) {
  // use hip - knee - ankle angle (left or right preferably best scored)
  const leftHip = kp(pose, "left_hip");
  const leftKnee = kp(pose, "left_knee");
  const leftAnkle = kp(pose, "left_ankle");
  const rightHip = kp(pose, "right_hip");
  const rightKnee = kp(pose, "right_knee");
  const rightAnkle = kp(pose, "right_ankle");

  let angle = null;
  if (leftHip?.score > 0.35 && leftKnee?.score > 0.35 && leftAnkle?.score > 0.35) {
    angle = calculateAngle(leftHip, leftKnee, leftAnkle);
  } else if (rightHip?.score > 0.35 && rightKnee?.score > 0.35 && rightAnkle?.score > 0.35) {
    angle = calculateAngle(rightHip, rightKnee, rightAnkle);
  } else return;

  state.lastAngle = angle;

  // thresholds
  const downThreshold = 95;    // angle < downThreshold => down
  const upThreshold = 155;     // angle > upThreshold => up

  // cues: if standing but not going low enough prompt "go lower"
  const now = Date.now();
  if (!state.down && angle > 120 && now - state.lastTimeCue > 1500) {
    cueEl.textContent = "Go lower";
    speak("Go lower");
    state.lastTimeCue = now;
  }

  if (angle < downThreshold && !state.down) {
    state.down = true;
    cueEl.textContent = "Down";
  }

  if (angle > upThreshold && state.down) {
    // counted a rep
    repCount++;
    countEl.textContent = repCount;
    state.down = false;
    cueEl.textContent = "Good! Stand up";
    speak("Good");
  }
}

function handlePushups(pose) {
  // elbow angle: shoulder - elbow - wrist
  const leftShoulder = kp(pose, "left_shoulder");
  const leftElbow = kp(pose, "left_elbow");
  const leftWrist = kp(pose, "left_wrist");
  const rightShoulder = kp(pose, "right_shoulder");
  const rightElbow = kp(pose, "right_elbow");
  const rightWrist = kp(pose, "right_wrist");

  let leftAngle = null, rightAngle = null;
  if (leftShoulder && leftElbow && leftWrist && leftElbow.score>0.35) {
    leftAngle = calculateAngle(leftShoulder, leftElbow, leftWrist);
  }
  if (rightShoulder && rightElbow && rightWrist && rightElbow.score>0.35) {
    rightAngle = calculateAngle(rightShoulder, rightElbow, rightWrist);
  }

  const angle = (leftAngle || rightAngle);
  if (!angle) return;
  state.lastAngle = angle;

  const downThreshold = 70;  // less than = down
  const upThreshold = 150;

  const now = Date.now();
  if (!state.down && angle > 110 && now - state.lastTimeCue > 1300) {
    cueEl.textContent = "Lower chest";
    speak("Lower your chest");
    state.lastTimeCue = now;
  }

  if (angle < downThreshold && !state.down) {
    state.down = true;
    cueEl.textContent = "Down";
  }
  if (angle > upThreshold && state.down) {
    repCount++;
    countEl.textContent = repCount;
    state.down = false;
    cueEl.textContent = "Good!";
    speak("Push up");
  }
}

function handleJumpingJacks(pose) {
  // measure distance between wrists relative to shoulder width
  const leftWrist = kp(pose, "left_wrist");
  const rightWrist = kp(pose, "right_wrist");
  const leftShoulder = kp(pose, "left_shoulder");
  const rightShoulder = kp(pose, "right_shoulder");
  if (!leftWrist || !rightWrist || !leftShoulder || !rightShoulder) return;
  if (leftWrist.score < 0.35 || rightWrist.score < 0.35) return;

  const wristDistance = dist(leftWrist, rightWrist);
  const shoulderDistance = dist(leftShoulder, rightShoulder);
  const norm = wristDistance / Math.max(shoulderDistance, 1);

  const openThreshold = 2.0; // when hands are far apart (above 2x shoulder width)
  const now = Date.now();

  if (!state.down && norm > openThreshold) {
    state.down = true; // open
    cueEl.textContent = "Jump!";
    speak("Jump");
  }
  if (state.down && norm < 1.2) {
    // closed -> count a rep
    repCount++;
    countEl.textContent = repCount;
    state.down = false;
    cueEl.textContent = "Nice!";
  }
}

function handleLunges(pose) {
  // detect front knee angle change: hip-knee-ankle of one leg
  const leftHip = kp(pose, "left_hip");
  const leftKnee = kp(pose, "left_knee");
  const leftAnkle = kp(pose, "left_ankle");
  const rightHip = kp(pose, "right_hip");
  const rightKnee = kp(pose, "right_knee");
  const rightAnkle = kp(pose, "right_ankle");

  // pick side with better keypoint scores
  let angle = null;
  if (leftHip?.score>0.35 && leftKnee?.score>0.35 && leftAnkle?.score>0.35) {
    angle = calculateAngle(leftHip, leftKnee, leftAnkle);
  } else if (rightHip?.score>0.35 && rightKnee?.score>0.35 && rightAnkle?.score>0.35) {
    angle = calculateAngle(rightHip, rightKnee, rightAnkle);
  } else return;

  const downThreshold = 100; // knee < 100 => stepping down
  const upThreshold = 160;

  const now = Date.now();
  if (!state.down && angle < downThreshold) {
    state.down = true;
    cueEl.textContent = "Lunge down";
  }
  if (state.down && angle > upThreshold) {
    repCount++;
    countEl.textContent = repCount;
    state.down = false;
    cueEl.textContent = "Return";
    speak("Good");
  }
}

// Main exercise dispatcher
function runExerciseLogic(name, pose) {
  switch (name) {
    case "squats": handleSquats(pose); break;
    case "pushups": handlePushups(pose); break;
    case "jumpingjacks": handleJumpingJacks(pose); break;
    case "lunges": handleLunges(pose); break;
    default: break;
  }
}

// --- Recommendations based on profile ---
function recommendExercises(profile) {

  const { age, weight, height, level } = profile;
  // very simple rule-based suggestion

  const strength = level === "beginner" ? "Bodyweight Squats, Push-ups (knees)" : "Squats, Push-ups, Lunges";
  const repsBase = level === "beginner" ? 8 : level === "intermediate" ? 12 : 16;
  const sets = level === "beginner" ? 2 : level === "intermediate" ? 3 : 4;

  const recText = `Recommended: ${strength}. Also: ${cardio}. Suggested: ${repsBase} reps × ${sets} sets. Adjust intensity as needed.`;
  recEl.textContent = recText;
  profileSummary.textContent = `Profile: ${age}y • ${height}cm • ${weight}kg • ${level}`;
}

// detection loop
async function detectLoop() {
  if (!detector) return;
  const posesNow = await detector.estimatePoses(video, { maxPoses: 1, flipHorizontal: false });
  ctx.clearRect(0,0,canvas.width,canvas.height);
  // draw video onto canvas as background so overlay aligns perfectly
  ctx.drawImage(video, 0, 0, canvas.width, canvas.height);

  const detectorConfig = { modelType: poseDetection.movenet.modelType.SINGLEPOSE_LIGHTNING };

// Detection loop
async function detectLoop() {
  const poses = await detector.estimatePoses(video);

  ctx.clearRect(0, 0, canvas.width, canvas.height);
  ctx.drawImage(video, 0, 0, canvas.width, canvas.height);

  if (poses.length > 0) {
    drawKeypoints(poses[0].keypoints);
    detectExercise(poses[0].keypoints);

  requestAnimationFrame(detectLoop);

  keypoints.forEach((k) => {

let video = document.getElementById("video");
let canvas = document.getElementById("canvas");
let ctx = canvas.getContext("2d");

let detector;
let currentExercise = "squats";
let repCount = 0;
let caloriesBurned = 0;
let statusText = "Waiting...";
let goingDown = false;

let user = { weight: 70, age: 25, fitness: "Intermediate" };

// 🎵 Audio feedback
function speak(text) {
  let msg = new SpeechSynthesisUtterance(text);
  msg.lang = "en-US";
  window.speechSynthesis.speak(msg);
}

// ✅ Setup camera
async function setupCamera() {
  try {
    const stream = await navigator.mediaDevices.getUserMedia({ video: true });
    video.srcObject = stream;
    return new Promise((resolve) => {
      video.onloadedmetadata = () => {
        video.play();
        resolve(video);
      };
    });
  } catch (err) {
    alert("Camera error: " + err.message);
    console.error(err);
  }
}

// ✅ Initialize detector
async function init() {
  await setupCamera();
  const detectorConfig = { modelType: poseDetection.movenet.modelType.SINGLEPOSE_LIGHTNING };
  detector = await poseDetection.createDetector(poseDetection.SupportedModels.MoveNet, detectorConfig);

  canvas.width = video.videoWidth;
  canvas.height = video.videoHeight;

  detectLoop();
}

// ✅ Main loop
async function detectLoop() {
  const poses = await detector.estimatePoses(video);
  ctx.clearRect(0, 0, canvas.width, canvas.height);
  ctx.drawImage(video, 0, 0, canvas.width, canvas.height);

  if (poses.length > 0) {
    drawKeypoints(poses[0].keypoints);
    detectExercise(poses[0].keypoints);
  }

  requestAnimationFrame(detectLoop);
}

// ✅ Draw pose dots
function drawKeypoints(keypoints) {
  keypoints.forEach((k) => {
    if (k.score > 0.5) {
      ctx.beginPath();
      ctx.arc(k.x, k.y, 5, 0, 2 * Math.PI);
      ctx.fillStyle = "red";
      ctx.fill();
    }
  });
}

// ✅ Exercise detection logic
function detectExercise(keypoints) {
  if (currentExercise === "squats") {
    const hip = keypoints.find(k => k.name === "left_hip");
    const knee = keypoints.find(k => k.name === "left_knee");

    if (hip && knee) {
      if (hip.y < knee.y) {
        if (!goingDown) {
          repCount++;
          caloriesBurned += calcCalories("squats");
          speak("Good rep!");
          goingDown = true;
        }
        statusText = "Standing";
      } else {
        goingDown = false;
        statusText = "Squatting";
      }
    }
  }

  if (currentExercise === "pushups") {
    const shoulder = keypoints.find(k => k.name === "left_shoulder");
    const elbow = keypoints.find(k => k.name === "left_elbow");

    if (shoulder && elbow) {
      if (shoulder.y < elbow.y) {
        if (!goingDown) {
          repCount++;
          caloriesBurned += calcCalories("pushups");
          speak("Nice push!");
          goingDown = true;
        }
        statusText = "Up";
      } else {
        goingDown = false;
        statusText = "Down";
      }
    }
  }

  if (currentExercise === "jumpingjacks") {
    const wrist = keypoints.find(k => k.name === "left_wrist");
    const ankle = keypoints.find(k => k.name === "left_ankle");

    if (wrist && ankle) {
      if (wrist.y < ankle.y) {
        if (!goingDown) {
          repCount++;
          caloriesBurned += calcCalories("jumpingjacks");
          speak("Keep it up!");
          goingDown = true;
        }
        statusText = "Jump!";
      } else {
        goingDown = false;
        statusText = "Ready";
      }
    }
  }

  document.getElementById("repCount").innerText = "Reps: " + repCount;
  document.getElementById("calories").innerText = "Calories: " + caloriesBurned.toFixed(2);
  document.getElementById("status").innerText = statusText;
}

// ✅ Simple calorie estimation
function calcCalories(exercise) {
  let factor = 0.5; // baseline multiplier
  if (exercise === "squats") factor = 0.32;
  if (exercise === "pushups") factor = 0.29;
  if (exercise === "jumpingjacks") factor = 0.20;
  return (user.weight / 70) * factor; // scale with weight
}

function calculateAngle(a, b, c) {
  const ab = [a.x - b.x, a.y - b.y];
  const cb = [c.x - b.x, c.y - b.y];

  const dot = ab[0] * cb[0] + ab[1] * cb[1];
  const magAB = Math.sqrt(ab[0] ** 2 + ab[1] ** 2);
  const magCB = Math.sqrt(cb[0] ** 2 + cb[1] ** 2);

// helper: calculate angle between 3 points
function getAngle(a, b, c) {
  const ab = { x: a.x - b.x, y: a.y - b.y };
  const cb = { x: c.x - b.x, y: c.y - b.y };
  const dot = ab.x * cb.x + ab.y * cb.y;
  const magAB = Math.sqrt(ab.x * ab.x + ab.y * ab.y);
  const magCB = Math.sqrt(cb.x * cb.x + cb.y * cb.y);
  return (Math.acos(dot / (magAB * magCB)) * 180) / Math.PI;
}

function drawSkeleton(keypoints, scaleX, scaleY) {
  const pairs = [
    ["left_shoulder", "right_shoulder"],
    ["left_hip", "right_hip"],
    ["left_shoulder", "left_hip"],
    ["right_shoulder", "right_hip"],
    ["left_hip", "left_knee"],
    ["right_hip", "right_knee"],
    ["left_knee", "left_ankle"],
    ["right_knee", "right_ankle"],
    ["left_elbow", "left_wrist"],
    ["right_elbow", "right_wrist"],
    ["left_shoulder", "left_elbow"],
    ["right_shoulder", "right_elbow"]
  ];

  ctx.strokeStyle = "rgba(255, 0, 0, 0.6)";
  ctx.lineWidth = 3;

  pairs.forEach(([a, b]) => {
    const p1 = keypoints.find(k => k.name === a && k.score > 0.4);
    const p2 = keypoints.find(k => k.name === b && k.score > 0.4);
    if (p1 && p2) {
      ctx.beginPath();
      ctx.moveTo(p1.x * scaleX, p1.y * scaleY);
      ctx.lineTo(p2.x * scaleX, p2.y * scaleY);
      ctx.stroke();
    }
  });

  // Draw joints
  keypoints.forEach(p => {
    if (p.score > 0.4) {
      ctx.beginPath();
      ctx.arc(p.x * scaleX, p.y * scaleY, 5, 0, 2 * Math.PI);
      ctx.fillStyle = "red";
      ctx.fill();
    }
  });
}

// ---- drawing utilities ----
function drawSkeletonAndKeypoints(keypoints, scaleX, scaleY) {
  // skeleton pairs
  const pairs = [
    ["left_shoulder","right_shoulder"],
    ["left_shoulder","left_elbow"],
    ["left_elbow","left_wrist"],
    ["right_shoulder","right_elbow"],
    ["right_elbow","right_wrist"],
    ["left_shoulder","left_hip"],
    ["right_shoulder","right_hip"],
    ["left_hip","right_hip"],
    ["left_hip","left_knee"],
    ["left_knee","left_ankle"],
    ["right_hip","right_knee"],
    ["right_knee","right_ankle"]
  ];

  ctx.lineWidth = 3;
  ctx.strokeStyle = "rgba(179,0,0,0.9)";

  pairs.forEach(([a,b]) => {
    const p1 = keypoints.find(k => (k.name===a || k.part===a) && k.score>0.35);
    const p2 = keypoints.find(k => (k.name===b || k.part===b) && k.score>0.35);
    if (p1 && p2) {
      ctx.beginPath();
      ctx.moveTo(p1.x * scaleX, p1.y * scaleY);
      ctx.lineTo(p2.x * scaleX, p2.y * scaleY);
      ctx.stroke();
    }
  });

  // improved keypoints - outer colored, inner white
  keypoints.forEach(p => {
    if (!p || p.score < 0.35) return;
    const x = p.x * scaleX, y = p.y * scaleY;
    // outer
    ctx.beginPath();
    ctx.fillStyle = "rgba(179,0,0,0.95)";
    ctx.arc(x, y, 6, 0, Math.PI*2);
    ctx.fill();
    // inner
    ctx.beginPath();
    ctx.fillStyle = "#fff";
    ctx.arc(x, y, 2.5, 0, Math.PI*2);
    ctx.fill();
  });
}

// ---- helper math ----
function mean(values) {
  if (!values.length) return 0;
  return values.reduce((s,v)=>s+v,0)/values.length;
}

// ---- detection loop ----
async function detectLoop() {
  if (!detector) return;
  const poses = await detector.estimatePoses(video, { maxPoses: 1, flipHorizontal: false });

  // ensure canvas mirrors video size properly
  const videoW = video.videoWidth || 1280;
  const videoH = video.videoHeight || 720;
  if (canvas.width !== videoW || canvas.height !== videoH) {
    canvas.width = videoW;
    canvas.height = videoH;
  }

  ctx.clearRect(0,0,canvas.width,canvas.height);
  // draw video frame to canvas
  ctx.drawImage(video, 0, 0, canvas.width, canvas.height);

  if (poses && poses.length > 0) {
    const pose = poses[0];
    const keypoints = pose.keypoints;

    const scaleX = canvas.width / videoW;
    const scaleY = canvas.height / videoH;
    drawSkeletonAndKeypoints(keypoints, scaleX, scaleY);

    // get required keypoints (use name or part for compatibility)
    const getKP = name => keypoints.find(k => (k.name === name || k.part === name));

    const nose = getKP("nose");
    const leftHip = getKP("left_hip");
    const rightHip = getKP("right_hip");
    const leftShoulder = getKP("left_shoulder");
    const rightShoulder = getKP("right_shoulder");
    const leftKnee = getKP("left_knee");
    const rightKnee = getKP("right_knee");

    // if essential points missing, skip heavy logic
    if (!nose || !leftHip || !rightHip || !leftShoulder || !rightShoulder) {
      statusEl.textContent = "Status: move into frame";
      requestAnimationFrame(detectLoop);
      return;
    }

    const exercise = exerciseSelect.value;

    // ------------------ SQUAT (KEEP EXACT METHOD YOU GAVE) ------------------
    if (exercise === "squat" && leftHip && rightHip && leftKnee && rightKnee) {
      // YOUR ORIGINAL SQUAT METHOD (unchanged)
      const torsoY = ((leftHip.y + rightHip.y) / 2 + nose.y) / 2;
      if (!initialTorsoY) initialTorsoY = torsoY;
      const movement = torsoY - initialTorsoY;

      if (movement > 50 && !squatDown) {
        squatDown = true;
        speakOnce("Down");
        statusEl.textContent = "Status: Squat Down";
      }
      if (movement < 20 && squatDown) {
        squatDown = false;
        squatCount++;
        squatCountEl.textContent = squatCount;
        speakOnce("Up. Good job");
        statusEl.textContent = "Status: Squat Up";
      }
    }

    // ------------------ PUSH-UP (NEW ROBUST METHOD) ------------------
    if (exercise === "pushup" && leftHip && rightHip && leftShoulder && rightShoulder && nose) {
      // compute averages
      const avgHipY = (leftHip.y + rightHip.y) / 2;
      const avgShoulderY = (leftShoulder.y + rightShoulder.y) / 2;
      const avgNoseY = nose.y;

      // bodyHeight relative measure (shoulder above hips)
      const bodyHeight = Math.max(1, avgShoulderY - avgHipY);

      // positive when nose is below shoulders (i.e., body lowered)
      const rawPushDepth = avgNoseY - avgShoulderY;

      // normalized depth ratio (0 = fully up, larger = deeper)
      const depthRatio = rawPushDepth / bodyHeight;

      // smoothing buffer
      pushDepthBuffer.push(depthRatio);
      if (pushDepthBuffer.length > PUSH_BUFFER_SIZE) pushDepthBuffer.shift();
      const smoothDepth = mean(pushDepthBuffer);

      // thresholds tuned for typical webcam positions
      const DOWN_THRESHOLD = 0.55; // when depth ratio exceeds this => down
      const UP_THRESHOLD = 0.25;   // when depth ratio below this => up

      // debug status
      statusEl.textContent = `Status: Push depth ${smoothDepth.toFixed(2)}`;

      // detect down
      if (smoothDepth > DOWN_THRESHOLD && !pushupDown) {
        pushupDown = true;
        speakOnce("Down");
        // small visual cue already via status
      }

      // detect up
      if (smoothDepth < UP_THRESHOLD && pushupDown) {
        pushupDown = false;
        pushupCount++;
        pushupCountEl.textContent = pushupCount;
        speakOnce("Up. Good push");
      }
    }
  } else {
    statusEl.textContent = "Status: No person detected";
  }

  requestAnimationFrame(detectLoop);
}

  synth.speak(utter);
}

    ["left_shoulder", "right_shoulder"],
    ["left_hip", "right_hip"],

  pairs.forEach(([a, b]) => {

    if (p1 && p2 && p1.score > 0.4 && p2.score > 0.4) {

function drawKeypointsAndSkeleton(keypoints){
  ctx.clearRect(0,0,canvas.width,canvas.height);
  ctx.drawImage(video, 0, 0, canvas.width, canvas.height);

  // skeleton pairs (use keypoint.name)
  const pairs = [
    ['left_shoulder','right_shoulder'],
    ['left_shoulder','left_elbow'],
    ['left_elbow','left_wrist'],
    ['right_shoulder','right_elbow'],
    ['right_elbow','right_wrist'],
    ['left_shoulder','left_hip'],
    ['right_shoulder','right_hip'],
    ['left_hip','right_hip'],
    ['left_hip','left_knee'],
    ['right_hip','right_knee'],
    ['left_knee','left_ankle'],
    ['right_knee','right_ankle']
  ];

  ctx.lineWidth = 4;
  ctx.strokeStyle = 'rgba(255,40,40,0.95)';
  ctx.fillStyle = 'rgba(255,40,40,0.97)';

  // draw lines
  pairs.forEach(([a,b])=>{
    const A = keypoints.find(k => k.name === a || k.part === a);
    const B = keypoints.find(k => k.name === b || k.part === b);
    if(A && B && A.score > 0.35 && B.score > 0.35){
      ctx.beginPath();
      ctx.moveTo(A.x, A.y);
      ctx.lineTo(B.x, B.y);
      ctx.stroke();
    }
  });

  // draw points: outer red, inner white
  keypoints.forEach(p=>{
    if(!p || p.score < 0.35) return;
    ctx.beginPath();
    ctx.fillStyle = 'rgba(179,0,0,0.95)'; ctx.arc(p.x, p.y, 6, 0, Math.PI*2); ctx.fill();
    ctx.beginPath(); ctx.fillStyle = '#fff'; ctx.arc(p.x, p.y, 2.5, 0, Math.PI*2); ctx.fill();
  });
}

// main detection loop
async function detectLoop(){
  if(!detector) return;
  const poses = await detector.estimatePoses(video, { maxPoses: 1, flipHorizontal: false });

  if(poses && poses.length > 0){
    const keypoints = poses[0].keypoints;
    drawKeypointsAndSkeleton(keypoints);

    const get = name => keypoints.find(k => k.name === name || k.part === name);

    const nose = get('nose');
    const leftHip = get('left_hip'), rightHip = get('right_hip');
    const leftElbow = get('left_elbow'), rightElbow = get('right_elbow');
    const leftKnee = get('left_knee'), rightKnee = get('right_knee');
    const leftShoulder = get('left_shoulder'), rightShoulder = get('right_shoulder');

    // if person not fully in frame, request move in
    if(!nose || !leftHip || !rightHip){ statusEl.textContent = 'Status: move into frame'; requestAnimationFrame(detectLoop); return; }

    const ex = exerciseSelect.value;

    // SQUAT — keep EXACT method (unchanged)
    if(ex === 'squat' && leftKnee && rightKnee){
      const torsoY = ((leftHip.y + rightHip.y)/2 + nose.y)/2;
      if(!initialTorsoY) initialTorsoY = torsoY;
      const movement = torsoY - initialTorsoY;

      if(movement > 50 && !squatDown){
        squatDown = true;
        speakOnce('Down');
        statusEl.textContent = 'Status: Squat Down';
      }
      if(movement < 20 && squatDown){
        squatDown = false;
        squatCount++;
        squatCountEl.textContent = squatCount;
        speakOnce('Up');
        statusEl.textContent = 'Status: Squat Up';
      }
    }

    // PUSH-UP — **side-view**: head (nose) + elbow motion (choose left elbow) — NO depth ratio
    if(ex === 'pushup' && leftElbow && rightElbow && leftShoulder && rightShoulder && nose){
      // choose side with higher elbow confidence
      const side = (leftElbow.score >= rightElbow.score) ? 'left' : 'right';
      const elbow = (side === 'left') ? leftElbow : rightElbow;

      const headY = nose.y;
      const elbowY = elbow.y;
      const avgY = (headY + elbowY) / 2;

      if(!initialElbowY) initialElbowY = avgY;
      const delta = avgY - initialElbowY;

      // tuning thresholds — px values; adjust if camera farther/closer
      const DOWN_T = 36;
      const UP_T = 14;

      statusEl.textContent = `Status: push delta ${delta.toFixed(1)}`;

      if(delta > DOWN_T && !pushupDown){
        pushupDown = true;
        speakOnce('Down');
      }
      if(delta < UP_T && pushupDown){
        pushupDown = false;
        pushupCount++;
        pushupCountEl.textContent = pushupCount;
        speakOnce('Up');
      }
    }

  } else {
    statusEl.textContent = 'Status: No person detected';
  }

  requestAnimationFrame(detectLoop);
}

async function loadPoseDetector() {
    return poseDetection.createDetector(

function angle(a, b, c) {
    let ab = Math.hypot(a.x - b.x, a.y - b.y);
    let bc = Math.hypot(b.x - c.x, b.y - c.y);
    let ac = Math.hypot(a.x - c.x, a.y - c.y);
    return Math.acos((ab**2 + bc**2 - ac**2) / (2 * ab * bc)) * (180 / Math.PI);
}

let squatDown = false;
let pushDown = false;

function countSquat(pose) {
    let hip = pose.keypoints[11];
    let knee = pose.keypoints[13];

    if (hip.score > .4 && knee.score > .4) {
        if (hip.y < knee.y) squatDown = true;
        if (squatDown && hip.y > knee.y) {
            squatCount++;
            document.getElementById("squatCount").textContent = squatCount;
            squatDown = false;
        }
    }
}

function countPushup(pose) {
    let head = pose.keypoints[0];
    let elbow = pose.keypoints[7];

    if (head.score > .4 && elbow.score > .4) {
        if (head.y < elbow.y - 40) pushDown = true;
        if (pushDown && head.y > elbow.y - 15) {
            pushupCount++;
            document.getElementById("pushupCount").textContent = pushupCount;
            pushDown = false;
        }
    }
}

async function run() {
    await startCamera();
    canvas.width = video.videoWidth;
    canvas.height = video.videoHeight;

    let detector = await loadPoseDetector();

    async function detect() {
        const poses = await detector.estimatePoses(video);
        ctx.drawImage(video, 0, 0, canvas.width, canvas.height);

        if (poses.length > 0) {
            let pose = poses[0];

            // draw skeleton
            pose.keypoints.forEach((p) => {
                ctx.fillStyle = "red";
                ctx.beginPath(); ctx.arc(p.x, p.y, 5, 0, 2 * Math.PI); ctx.fill();
            });

            if (currentExercise === "squat") countSquat(pose);
            else if (currentExercise === "pushup") countPushup(pose);
        }

        requestAnimationFrame(detect);
    }
    detect();
}

function drawSkeleton(keypoints, scaleX, scaleY) {
  const pairs = [
    ["left_shoulder", "right_shoulder"],
    ["left_hip", "right_hip"],
    ["left_shoulder", "left_hip"],
    ["right_shoulder", "right_hip"],
    ["left_hip", "left_knee"],
    ["right_hip", "right_knee"],
    ["left_knee", "left_ankle"],
    ["right_knee", "right_ankle"],
    ["left_elbow", "left_wrist"],
    ["right_elbow", "right_wrist"],
    ["left_shoulder", "left_elbow"],
    ["right_shoulder", "right_elbow"],
    ["nose", "left_eye"],
    ["nose", "right_eye"]
  ];

  ctx.strokeStyle = "rgba(255,0,0,0.6)";
  ctx.lineWidth = 3;

  pairs.forEach(([a, b]) => {
    const p1 = keypoints.find(k => k.name === a && k.score > 0.4);
    const p2 = keypoints.find(k => k.name === b && k.score > 0.4);
    if (p1 && p2) {
      ctx.beginPath();
      ctx.moveTo(p1.x * scaleX, p1.y * scaleY);
      ctx.lineTo(p2.x * scaleX, p2.y * scaleY);
      ctx.stroke();
    }
  });

  // draw red joints
  keypoints.forEach(p => {
    if (p.score > 0.4) {
      ctx.beginPath();
      ctx.arc(p.x * scaleX, p.y * scaleY, 6, 0, 2 * Math.PI);
      ctx.fillStyle = "red";
      ctx.fill();
    }
  });
}

// helper: calculate angle between 3 points
function getAngle(a, b, c) {
  const ab = { x: a.x - b.x, y: a.y - b.y };
  const cb = { x: c.x - b.x, y: c.y - b.y };
  const dot = ab.x * cb.x + ab.y * cb.y;
  const magAB = Math.sqrt(ab.x * ab.x + ab.y * ab.y);
  const magCB = Math.sqrt(cb.x * cb.x + cb.y * cb.y);
  return (Math.acos(dot / (magAB * magCB)) * 180) / Math.PI;
}

          pushupCount++;
          pushupCountEl.textContent = pushupCount;
          speak("Up");
        }
      }
    }
  }
